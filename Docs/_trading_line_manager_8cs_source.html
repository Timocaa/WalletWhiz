134 private DateTime GetNextDate(DateTime date)
135 {
136 DateTime today = DateTime.Today;
137 // If the transaction date is in the future or no periodicity is defined, return the original date.
138 if (today < date || string.IsNullOrEmpty(_periodicity))
139 return date;
140 // Increment the date until it is after today.
141 while (date <= today)
142 {
143 switch (_periodicity)
144 {
145 case "Day":
146 date = date.AddDays(1);
147 break;
148 case "Week":
149 date = date.AddDays(7);
150 break;
151 case "Month":
152 date = date.AddMonths(1);
153 break;
154 case "Year":
155 date = date.AddYears(1);
156 break;
157 }
158 }
159 return date;
160 }
