150 private bool FormIsFull()
151 {
152 // Ensure the amount field is not empty.
153 if (string.IsNullOrEmpty(amountText.text))
154 return false;
155 // Ensure that exactly one of the toggles is selected.
156 if ((expenseToggle.isOn && incomeToggle.isOn) || (!incomeToggle.isOn && !expenseToggle.isOn))
157 return false;
158 // Ensure the description field is not empty.
159 if (string.IsNullOrEmpty(describeText.text))
160 return false;
161 // If a periodicity is selected (i.e., recurring transaction), adjust the description to make it unique.
162 if (periodicityDropdown.value > 0)
163 {
164 List<Transaction> transactions = databaseManager.GetTransactions();
165 int occurence = 1;
166 foreach (Transaction trade in transactions)
167 {
168 // Process only recurring transactions.
169 if (trade.isRehearsals)
170 {
171 // If the existing trade description matches the entered description exactly.
172 if (string.Equals(trade.describe, describeText.text, StringComparison.Ordinal))
173 {
174 // If the description already ends with the occurrence indicator, remove it.
175 if (describeText.text.EndsWith("#" + occurence))
176 describeText.text =
177 describeText.text.Remove(describeText.text.Length - ("#" + occurence).Length);
178 // Append the occurrence indicator.
179 describeText.text = describeText.text + "#" + occurence;
180 occurence++;
181 }
182 }
183 }
184 }
185 return true;
