146 private void GetAllCategoryTrades(List<Transaction> transactions)
147 {
148 foreach (var trade in transactions)
149 {
150 // If no groups exist, create the first group.
151 if (_categoryTradesList.Count == 0)
152 {
153 List<Transaction> newList = new List<Transaction> { trade };
154 _categoryTradesList.Add(newList);
155 }
156 else
157 {
158 int i = 0;
159 // Look for an existing group matching the transaction's category.
160 while (i < _categoryTradesList.Count)
161 {
162 if (string.Equals(_categoryTradesList[i][0].category, trade.category, StringComparison.Ordinal))
163 {
164 _categoryTradesList[i].Add(trade);
165 break;
166 }
167 i++;
168 }
169 // If no matching group is found, create a new group.
170 if (i == _categoryTradesList.Count)
171 {
172 List<Transaction> newList = new List<Transaction> { trade };
173 _categoryTradesList.Add(newList);
174 }
175 }
176 }
177 }

217 private void CalculateAmounts()
218 {
219 _expenseAmount = 0f;
220 _incomeAmount = 0f;
221 _subtotals.Clear();
222 // Iterate through each group of transactions.
223 foreach (var categoryTrade in _categoryTradesList)
224 {
225 float subtotal = 0f;
226 foreach (var trade in categoryTrade)
227 {
228 // Sum up expenses and incomes.
229 if (string.Equals(trade.type, "expense", StringComparison.Ordinal))
230 _expenseAmount += trade.amount;
231 else
232 _incomeAmount += trade.amount;
233 subtotal += trade.amount;
234 }
235 _subtotals.Add(subtotal);
236 }
237 _totalAmount = _expenseAmount + _incomeAmount;
238 }

244 private void DisplaySlices(string category)
245 {
246 // Calculate the amounts and subtotals.
247 CalculateAmounts();
248 float currentRotation = 0f;
249 for (int i = 0; i < _subtotals.Count ; i++)
250 {
251 // Determine the fraction of the total for the current category.
252 float fraction = _subtotals[i] / _totalAmount;
253 float sliceAngle = fraction * 360f;
254 // Instantiate a slice from the prefab.
255 RectTransform sliceInstance = Instantiate(slicePrefab, graphArea.transform);
256 _sliceObjects.Add(sliceInstance);
257 sliceInstance.anchoredPosition = Vector2.zero;
258 sliceInstance.localRotation = Quaternion.Euler(0, 0, -currentRotation);
259 Image sliceImage = sliceInstance.GetComponent<Image>();
260 if (sliceImage != null)
261 {
262 // Set the slice's fill amount and color.
263 sliceImage.fillAmount = fraction;
264 sliceImage.color = string.Equals(category, "Toutes les catégories", StringComparison.Ordinal) ? sliceColors[i % sliceColors.Count] : oneCategoryColors[i % oneCategoryColors.Count];
265 }
266 currentRotation += sliceAngle;
267 }
