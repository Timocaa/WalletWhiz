177 private List<Transaction> GetExistingTrades(DateTime startDate, DateTime endDate)
178 {
179 List<Transaction> trades = databaseManager.GetTransactions();
180 List<Transaction> tradesFound = new List<Transaction>();
181 // Filter transactions by date.
182 foreach (var trade in trades)
183 {
184 DateTime.TryParseExact(trade.date, DateFormats, FrenchCulture, DateTimeStyles.None, out DateTime dateFormated);
185 if (dateFormated >= startDate && dateFormated <= endDate)
186 tradesFound.Add(trade);
187 }
188 return tradesFound;

196 private List<Transaction> GetTradesWithProvisional(DateTime endDate)
197 {
198 List<Transaction> transactions = databaseManager.GetTransactions();
199 List<Transaction> rehearsalsTrades = databaseManager.GetRehearsalsTrades();
200 // Process each recurring transaction.
201 foreach (var rehearsalElement in rehearsalsTrades)
202 {
203 // Build the description marker for recurring transactions.
204 string toCompare = rehearsalElement.describe + "#Répétition#";
205 // Determine the last known date of the recurring transaction.
206 DateTime.TryParseExact(rehearsalElement.date, DateFormats, FrenchCulture, DateTimeStyles.None, out DateTime lastDateConverted);
207 foreach (var trade in transactions)
208 {
209 if (String.Equals(toCompare, trade.describe, StringComparison.Ordinal))
210 {
211 DateTime.TryParseExact(trade.date, DateFormats, FrenchCulture, DateTimeStyles.None, out DateTime dateConverted);
212 if (dateConverted > lastDateConverted)
213 lastDateConverted = dateConverted;
214 }
215 }
216 // Calculate the next occurrence date based on the periodicity.
217 DateTime nextDate = GetNextDate(lastDateConverted, rehearsalElement.rehearsalsPeriod);
218 // Generate provisional transactions until the end date.
219 while (nextDate <= endDate)
220 {
221 // Create a provisional transaction.
222 Transaction t = GenerateTransaction(rehearsalElement, nextDate);
223 // Add the provisional transaction to the list.
224 transactions.Add(t);
225 // Calculate the next occurrence date.
226 nextDate = GetNextDate(nextDate, rehearsalElement.rehearsalsPeriod);
227 }
228 }
229 return transactions;
230 }

241 private Transaction GenerateTransaction(Transaction tradeToCopy, DateTime nextDate)
242 {
243 // Append the repetition marker to the original description to indicate it's a recurring transaction.
244 string describe = tradeToCopy.describe + "#Répétition#";
245 // Format the next date using the French culture format.
246 string newDate = nextDate.ToString("dddd dd MMMM yyyy", FrenchCulture);
247 // Create a provisional transaction with an id of -1 to denote that it's not stored in the database.
248 Transaction t = new Transaction
249 {
250 id = -1,
251 amount = tradeToCopy.amount,
252 category = tradeToCopy.category,
253 type = tradeToCopy.type,
254 date = newDate,
255 isRehearsals = false,
256 rehearsalsPeriod = "",
257 describe = describe,
258 };
259 return t;

268 private DateTime GetNextDate(DateTime date, string periodicity)
269 {
270 switch (periodicity)
271 {
272 case "Day":
273 return date.AddDays(1);
274 case "Week":
275 return date.AddDays(7);
276 case "Month":
277 return date.AddMonths(1);
278 case "Year":
279 return date.AddYears(1);

291 private List<Transaction> GetTransactionsDateToDate(string startDate, string endDate)
292 {
293 // Parse the start and end dates.
294 DateTime.TryParseExact(startDate, DateFormats, FrenchCulture, DateTimeStyles.None, out DateTime startDateFormated);
295 DateTime.TryParseExact(endDate, DateFormats, FrenchCulture, DateTimeStyles.None, out DateTime endDateFormated);
296 List<Transaction> transactions;
297 // If the end date is in the future, include provisional recurring transactions.
298 if (endDateFormated > DateTime.Today)
299 transactions = GetTradesWithProvisional(endDateFormated);
300 else
301 transactions = GetExistingTrades(startDateFormated, endDateFormated);
302 return transactions;
